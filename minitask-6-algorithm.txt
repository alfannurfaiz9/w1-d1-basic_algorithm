FUNCTION hitungTotalDiskon
    DECLARE pembelian, totalHarga : INTEGER
            diskon : FLOAT
BEGIN
    INPUT(pembelian)
    IF pembelian >= 500.000 THEN
        IF pembelian <= 1.000.000 THEN
            diskon <- 0.05
        ELSE 
            diskon <- 0.1
    ELSE 
        diskon <- 0
    END IF

    totalHarga <- pembelian - (pembelian * diskon)
    OUTPUT("totalHarga =", totalHarga)
END
